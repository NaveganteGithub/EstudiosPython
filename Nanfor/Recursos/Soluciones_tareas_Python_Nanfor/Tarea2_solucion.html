    """Devuelve una serie con las notas de los alumnos aprobados en orden descendente"""

    print(serie_notas[serie_notas >= 5].sort_values(ascending = False))

import pandas as pd

def caracterizar_csv(archivo):
    """Crea un DataFrame desde un archivo csv y devuelve otro DataFrame 
    con el mínimo, el máximo y la media de cada columna"""
    # por convención llamamos df a nuestro dataframe ya que las modificaciones son "vistas" del mismo
    # el método read_csv lee el contenido de un archivo 
    df = pd.read_csv(archivo, sep=";", decimal = ",") 
    # visualizamos el df creado, utilizando el parametro sep=";" para separar los valores del archivo.csv y "," como separador decimal
    print(df)
    # el metodo describe nos da un resumen de los valores de cada columna que no se ajusta a lo que pide el enunciado
    print(df.describe())
    # la columna volumen no la calcula bien por los puntos separadores de miles y millones, la dejamos fuera del analisis
    lista_columnas = ['Final','Máximo','Mínimo','Efectivo'] 
    # Mediante un bucle que recorre las columnas de la lista calculamos el mínimo, el máximo y la media de cada una y las imprimimos
    for columna in lista_columnas:     
        print(f"Columna {columna}:\nMínimo: {df[columna].min()}  Máximo: {df[columna].max()}  Media: {df[columna].mean()}") 
    return

import pandas as pd

def media_de_precio_por_fabricante(archivo):
    """Lee el archivo Automobile_data.csv, agrupa por fabricante y calcula la media de precio"""
    # convierte el archivo en un dataframe
    df = pd.read_csv(archivo) 
    # visualizamos el df original
    print(df) 
    # imprimimos una vista del df agrupando las filas por 'company' y calculando la media con el metodo mean() de la columna seleccionada 'price'
    print(df.groupby('company')['price'].mean())   
    return
